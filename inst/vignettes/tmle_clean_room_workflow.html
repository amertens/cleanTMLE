Clean-Room TMLE Workflow

cleanTMLE

2026-04-01

Overview

This vignette demonstrates the full clean-room staged TMLE workflow using the run_clean_tmle() wrapper. The staged workflow enforces outcome blinding through structured gates:

Stage Purpose Output
1 Lock analytic specification cleanroom_lock
2a Propensity-score diagnostics & gating PS diagnostics, ESS, love plot
2b Negative controls / plasmode simulation Bias, RMSE, coverage
3 Primary TMLE estimation ATE, IC, weight diagnostics

Simulated Data

We simulate a dataset with a binary treatment, baseline covariates, and a binary 24-month event outcome.

library(cleanTMLE)
#> cleanTMLE: Causal Inference for Risk Estimation
#> Use sim_func1() to generate example data, or data(example1) if available.
set.seed(42)

# Simulate via built-in DGP (or manually if sim_func1 not available)
n <- 800
age       <- rnorm(n, 55, 10)
sex       <- rbinom(n, 1, 0.45)
biomarker <- rnorm(n, 0, 1)
comorbid  <- rbinom(n, 1, 0.30)

# Treatment mechanism (confounded)
lp_trt <- -1.0 + 0.02 * age + 0.3 * sex + 0.4 * biomarker + 0.5 * comorbid
ps_true <- plogis(lp_trt)
treatment <- rbinom(n, 1, ps_true)

# Outcome mechanism (binary at 24 months)
lp_out <- -2.0 + 0.01 * age + 0.2 * sex + 0.3 * biomarker +
          0.4 * comorbid - 0.5 * treatment
event_24 <- rbinom(n, 1, plogis(lp_out))

# Negative-control outcome (unrelated to treatment)
nc_outcome <- rbinom(n, 1, plogis(-1.5 + 0.01 * age))

dat <- data.frame(
  age, sex, biomarker, comorbid, treatment, event_24, nc_outcome
)

Stage 1: Analysis Lock

The wrapper validates that a pre-specified SuperLearner library and truncation rules are provided, then creates a locked specification.

# These would normally be pre-registered
sl_lib <- c("SL.glm", "SL.mean")
trunc  <- c(0.01, 0.99)

Full Pipeline via run_clean_tmle()

result <- run_clean_tmle(
  data                 = dat,
  Avar                 = "treatment",
  Yvar                 = "event_24",
  covariates           = c("age", "sex", "biomarker", "comorbid"),
  learner_lib          = sl_lib,
  truncation           = trunc,
  neg_control_outcomes = "nc_outcome",
  simulation_spec      = list(reps = 20, effect_sizes = c(0.05)),
  stage2a_thresholds   = list(ESS = 0.20, max_smd = 0.25),
  seed                 = 42,
  verbose              = TRUE
)

Comparison with IPW and Matching

To illustrate improvements, we compare the TMLE estimate against crude, IPTW, and matching approaches using the cleanTMLE infrastructure.

# Create lock for comparison workflows
lock <- create_analysis_lock(
  data       = dat,
  treatment  = "treatment",
  outcome    = "event_24",
  covariates = c("age", "sex", "biomarker", "comorbid"),
  seed       = 42
)

ps_fit  <- fit_ps_glm(lock)
iptw    <- run_iptw_workflow(lock, ps_fit, override_clean_room = TRUE)
matched <- run_match_workflow(lock, ps_fit, override_clean_room = TRUE)

# Crude
Y <- dat$event_24; A <- dat$treatment
crude_rd <- mean(Y[A == 1]) - mean(Y[A == 0])

comparison <- data.frame(
  Method   = c("Crude", "IPTW", "PS Matching", "TMLE"),
  Estimate = round(c(crude_rd, iptw$estimate, matched$estimate,
                     result$tmle_result$estimates$ATE$estimate), 5),
  SE       = round(c(NA, iptw$se, matched$se,
                     result$tmle_result$estimates$ATE$se), 5),
  stringsAsFactors = FALSE
)
print(comparison)
